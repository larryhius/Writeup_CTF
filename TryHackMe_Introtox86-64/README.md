# Important stuff

```
# open the binary in debugging mode
r2 -d <file>

# analyze all symbols and entry points (common analysis command)
aa

# set the disassembly syntax to AT&T
e asm.syntax=att

# list function
afl

# print disassembly function
pdf @<func>

# what the function is used
iz
```
